    def next
      str = defined?(::SecureRandom) ? next_secure : next_md5
      packed = [str].pack('m')
      packed.tr!("=\r\n", '')
      packed.tr!('+/', '-_')
      packed
    end

    def next_md5
      now = Time.now
      dig = Digest::MD5.new
      dig.update(now.to_s)
      dig.update(now.usec.to_s)
      dig.update(rand(0).to_s)
      dig.update($$.to_s)
      dig.update(@salt.to_s)
      dig.digest
    end
