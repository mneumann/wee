  def next
    now = Time.now
    md5 = Digest::MD5.new
    md5.update(now.to_s)
    md5.update(now.usec.to_s)
    md5.update(rand(0).to_s)
    md5.update($$.to_s)
    md5.update(@salt.to_s)
    md5.hexdigest
  end
